# === Пример 11: Попытка обращения к неинициализированной переменной ===
# Использование значения, которому не присвоено начальное выражение

# Ошибка: использование неинициализированной переменной
result = uninitialized_var

# Ошибка: использование переменной до присваивания
x = y + 10  # y не инициализирована
y = 5

# Ошибка: использование переменной в функции до её инициализации
func test():
    a = b  # b не инициализирована
    b = 10

# Ошибка: использование переменной в условии до инициализации
if uninitialized:
    print("test")
