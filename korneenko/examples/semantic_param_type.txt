# === Пример 6: Тип фактических параметров не совпадает с ожидаемыми типами формальных ===
func process_int(x: int) -> int:
    return x * 2

func process_table(t: table) -> table:
    return t

func process_string(s: str) -> str:
    return s

# Ошибка: передача строки вместо int
result1 = process_int("10")

# Ошибка: передача числа вместо table
my_table = Table(id = Сolumn(int))
result2 = process_table(42)

# Ошибка: передача числа вместо str

# Ошибка: передача таблицы вместо int
