# Подпрограмма с возвратом  значений
func get_max(columnData: column[float]) -> float:
    maxVal = max(columnData)
    return maxVal


# Перегрузка по типу второго параметра
func filter_by_age(tableData: table, maxAge: int) -> table:
    result = [r for r in tableData if r.age < maxAge]
    maxAge = (maxAge + maxAge)*2
    return result
