# === Правильный пример 5: Switch и множественный выбор ===

func get_status_message(status: str) -> str:
    switch(status):
        case "new":
            return "Новый заказ"
        case "processing":
            return "В обработке"
        case "shipped":
            return "Отправлен"
        case "delivered":
            return "Доставлен"
        default:
            return "Неизвестный статус"

# Использование switch
order_status = "processing"
message = get_status_message(order_status)
print("Статус заказа:", message)

# Switch с числами
func get_day_name(day: int) -> str:
    switch(day):
        case 1:
            return "Понедельник"
        case 2:
            return "Вторник"
        case 3:
            return "Среда"
        case 4:
            return "Четверг"
        case 5:
            return "Пятница"
        case 6:
            return "Суббота"
        case 7:
            return "Воскресенье"
        default:
            return "Неверный день"

day_name = get_day_name(3)
print("День недели:", day_name)
