# === Правильный пример 1: Работа с переменными и типами ===

# Объявление переменных с типами
x: int = 10
y: float = 3.14
name: str = "Alice"
is_active: bool = true

# Присваивание без типа (автоматическое определение)
counter = 0
price = 99.99
message = "Hello, World!"

# Арифметические операции
sum = x + 5
product = x * y
division = sum / 2

# Логические операции
result = is_active and (x > 5)
condition = x == 10 or y < 5.0

# Вывод значений
print("Значение x:", x)
print("Сумма:", sum)
print("Результат:", result)
