# === Пример 10: Неверный порядок вычислений (побочные эффекты) ===
# Порядок вычислений нарушает ожидаемую семантику

x = 10
# Ошибка: изменение переменной до её использования в выражении
result = (x = 20) + x  # x изменяется в процессе вычисления

y = 5
# Ошибка: побочный эффект в выражении
value = y++ + y  # y изменяется между использованиями

z = 1
# Ошибка: изменение переменной в выражении
sum = z + (z = z + 1)  # z изменяется в процессе вычисления
