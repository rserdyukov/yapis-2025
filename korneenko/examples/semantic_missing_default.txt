# === Пример 8: Отсутствие ветви по умолчанию при обязательности default ===
# В многовариантном операторе не указано действие для неучтённых случаев

x = "test"

switch(x):
    case "one":
        print("1")
    case "two":
        print("2")
    # Ошибка: отсутствует default для неучтённых случаев

status = "unknown"

switch(status):
    case "new":
        return "Новый"
    case "processing":
        return "В обработке"
    # Ошибка: нет default для других значений
