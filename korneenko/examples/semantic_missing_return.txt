# === Пример 7: Подпрограмма должна возвращать значение, но оператор return отсутствует ===
func must_return_int() -> int:
    x = 10
    # Ошибка: функция должна возвращать int, но return отсутствует

func must_return_str() -> str:
    y = "test"
    # Ошибка: функция должна возвращать str, но return отсутствует

func no_return() -> void:
    z = 5
    # Это нормально: void функции могут не иметь return

func correct_return() -> int:
    return 42  # Правильно
