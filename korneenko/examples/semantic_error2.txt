# === Пример: Переопределение переменной ===
x = 10
x = 20  # Это нормально
x: int = 30  # Ошибка: переменная x уже объявлена

# === Пример: Использование переменной до объявления в функции ===
func test():
    y = x  # Ошибка: x не объявлена
    x = 10

# === Пример: Вызов функции с неправильными типами ===
func process_table(t: table) -> table:
    return t

my_var = 5
result = process_table(my_var)  # Ошибка: my_var не является таблицей

# === Пример: Return вне функции ===
x = 10
return x  # Ошибка: return вне функции
