# === Правильный пример 6: Списки и генераторы ===

# Создание списка
numbers = [1, 2, 3, 4, 5]
print("Список чисел:", numbers)

# Пустой список
empty_list = []
print("Пустой список:", empty_list)

# Генератор списка (list comprehension)
squares = [x * x for x in numbers]
print("Квадраты:", squares)

# Генератор с условием
even_squares = [x * x for x in numbers if x % 2 == 0]
print("Квадраты четных:", even_squares)

# Работа со списками
sum = 0
for num in numbers:
    sum = sum + num
print("Сумма:", sum)

# Функция для работы со списками
func sum_list(nums: list[int]) -> int:
    total = 0
    for n in nums:
        total = total + n
    return total

numbers_list = [10, 20, 30]
result = sum_list(numbers_list)
print("Сумма списка:", result)
