# === Пример 3: Неверное количество аргументов ===

    return a + b

result = test(10)  # Ошибка: ожидается 2 аргумента, получен 1

x += 5  # Ошибка: операция '+=' применима только к таблицам

# === Пример 5: Обращение к члену не-таблицы ===
