# === Правильный пример 3: Функции и возвращаемые значения ===

# Функция без параметров
func greet():
    print("Привет!")

# Функция с параметрами
func add(a: int, b: int) -> int:
    result = a + b
    return result

# Функция с одним параметром
func square(x: int) -> int:
    return x * x

# Функция без возвращаемого значения
func print_info(name: str):
    print("Имя:", name)

# Вызов функций
greet()
sum_result = add(5, 3)
squared = square(4)
print_info("Bob")

print("Сумма:", sum_result)
print("Квадрат:", squared)

# Функция с перегрузкой (разные сигнатуры)
func multiply(a: int, b: int) -> int:
    return a * b

func multiply(a: float, b: float) -> float:
    return a * b

result1 = multiply(3, 4)
result2 = multiply(2.5, 3.0)
print("Целое умножение:", result1)
print("Дробное умножение:", result2)
