set unique(set numbers):
    result = set[]

    i = 0
    while (i < size(numbers)):
        value = numbers[i]

        // проверяем, есть ли уже value в result
        found = false
        j = 0
        while (j < size(result)):
            if (result[j] == value):
                found = true
            j++

        if (!found):
            add(result, value)

        i++

    return result
