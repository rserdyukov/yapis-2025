func area_triangle(float a, float b, float angle_deg) -> float {
    float angle_rad = deg_to_rad(angle_deg);  # Преобразование градусов в радианы
    float sin_val = sin(angle_rad);           # Вычисление синуса угла
    return (a * b * sin_val) / 2.0;           # Формула площади
}

func main() -> void {
    float side1 = str_to_float(input());  # Ввод стороны a
    float side2 = str_to_float(input());  # Ввод стороны b
    float angle = str_to_float(input());  # Ввод угла в градусах
    str area = (str) area_triangle(side1, side2, angle);
    print(area);
}

main();
