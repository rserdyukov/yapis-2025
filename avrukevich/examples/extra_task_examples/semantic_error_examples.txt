# Семантические ошибки


# --- Ошибки несоответствия типов ---

# Ошибка: Смешивание типов в литерале массива.
# Массив должен содержать элементы только одного типа.
int[] mixed_type_array = [1, 2.5, 3]; # Несоответствие: int и float.

# Ошибка: Присваивание элемента неправильного типа в существующий массив.
str[] string_array = ["a", "b", "c"];
string_array[0] = 123; # Несоответствие: ожидается str, получен int.


# --- Ошибки неверных операций ---

# Ошибка: Попытка вызвать метод массива у переменной, не являющейся массивом.
int not_an_array = 10;
not_an_array.append(5); # У типа 'int' нет метода 'append'.

# Ошибка: Использование оператора, не определенного для массивов.
# Нельзя складывать (+) два массива.
int[] arr1 = [1, 2];
int[] arr2 = [3, 4];
int[] result = arr1 + arr2; # Оператор '+' не определен для типов int[] и int[].

# Ошибка: Использование массива в качестве условия, где ожидается bool.
if (arr1) then {
    print("This should not work"); # Условие должно быть типа bool, а не int[].
}


# --- Ошибки индексации ---

# Ошибка: Использование нецелочисленного типа (float) в качестве индекса.
arr1[0.5] = 1; # Индекс массива должен быть типа int.

# Ошибка: Использование нецелочисленного типа (string) в качестве индекса.
arr1["zero"] = 1; # Индекс массива должен быть типа int.


# --- Ошибки при передаче в функции ---

func takes_int(int x) -> void {
    print("Received int");
}

int[] array_arg = [100];
takes_int(array_arg); # Ошибка: функция ожидает int, но получает int[].
