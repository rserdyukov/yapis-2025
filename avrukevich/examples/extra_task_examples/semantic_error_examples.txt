# --- 1. Ошибки при создании и инициализации ---

# Ошибка: Несовпадение типов в литерале массива (int и float)
int[] mismatched_literal = [1, 2, 3.14, 4];

# Ошибка: Размер массива при создании через 'new' должен быть int, а не float
float[] new_with_float_size = new float[5.0];

# Ошибка: Пустой литерал массива не поддерживается, так как тип не может быть выведен
int[] empty_literal = [];


# --- 2. Ошибки доступа и использования свойств ---

int scalar_var = 100;
int[] my_array = [1, 2, 3];

# Ошибка: Попытка доступа по индексу к скалярной переменной
int invalid_access = scalar_var[0];

# Ошибка: Индекс массива должен быть int, а не float
int invalid_index = my_array[1.5];

# Ошибка: Свойство .length доступно только для массивов, но не для int
int invalid_length = scalar_var.length;


# --- 3. Ошибки присваивания ---

int[] i_arr_target = [0, 0];
float[] f_arr_source = [1.0, 2.0];
int scalar_target = 0;

# Ошибка: Нельзя присвоить массив скалярной переменной
scalar_target = i_arr_target;

# Ошибка: Нельзя присвоить скаляр переменной типа "массив"
i_arr_target = scalar_target;

# Ошибка: Нельзя присвоить массив float[] переменной типа int[]
i_arr_target = f_arr_source;

# Ошибка: Нельзя присвоить элементу массива int значение типа float
i_arr_target[0] = 3.14;


# --- 4. Ошибки сравнения (>, >=, <, <=, ==, !=) ---

int[] i_arr1 = [1, 2, 3];
int[] i_arr2 = [4, 5];
float[] f_arr1 = [1.0, 2.0, 3.0];
int scalar = 5;

# Ошибка: Нельзя сравнивать массив и скаляр с помощью операторов >, <, и т.д.
bool cmp_err1 = i_arr1 > scalar;

# Ошибка: Операторы == и != требуют, чтобы типы массивов были одинаковыми
bool cmp_err2 = i_arr1 == f_arr1;


# --- 5. Ошибки арифметических операций (Array vs Array) ---

int[] ia = [1, 2];
float[] fa = [3.0, 4.0];
str[] sa = ["a", "b"];
str[] sb = ["c", "d"];

# Ошибка: Несовпадение типов элементов (int[] + float[])
int[] arithm_err1 = ia + fa;

# Ошибка: Оператор '+' не поддерживается для массивов строк
str[] arithm_err2 = sa + sb;

# Ошибка: Оператор '*' не поддерживается для массивов строк
str[] arithm_err3 = sa * sb;


# --- 6. Ошибки арифметических операций (Array vs Scalar) ---

int[] i_arr_scalar_op = [10, 20];
str[] s_arr_scalar_op = ["hello"];
float scalar_f = 2.5;

# Ошибка: Несовпадение типов для скалярной операции (int[] * float)
int[] scalar_op_err1 = i_arr_scalar_op * scalar_f;

# Ошибка: Скалярные операции не поддерживаются для массивов строк
str[] scalar_op_err2 = s_arr_scalar_op * 2;


# --- 7. Ошибки арифметических присваиваний (+=, -=, ...) ---

int[] i_arr_assign = [1, 2];
float[] f_arr_assign = [3.0, 4.0];
str[] s_arr_assign = ["a", "b"];
int scalar_assign = 5;
float scalar_f_assign = 2.5;

# Ошибка: Несовпадение типов элементов (int[] += float[])
i_arr_assign += f_arr_assign;

# Ошибка: Оператор '+=' не поддерживается для массивов строк
s_arr_assign += s_arr_assign;

# Ошибка: Несовпадение типов для скалярного присваивания (int[] /= float)
i_arr_assign /= scalar_f_assign;

# Ошибка: Скалярное присваивание не поддерживается для массивов строк
s_arr_assign *= 2;


# --- 8. Ошибки порядка операндов (Scalar vs Array) ---

int[] i_arr_order = [1, 2];
int scalar_order = 10;

# Ошибка: Левый операнд должен быть массивом для операции "скаляр-массив".
# Операция 10 + [1, 2] не поддерживается, только [1, 2] + 10.
int[] order_error = scalar_order + i_arr_order;
scalar_order += i_arr_order;
