# Синтаксические шибки

# --- Ошибки в объявлении и инициализации ---

# Ошибка: Использование фігурних скобок {} вместо квадратных [] для инициализации.
int[] a = {1, 2, 3};

# Ошибка: Висячая запятая в конце списка инициализации.
# Грамматика ожидает выражение после каждой запятой.
float[] b = [1.0, 2.0, 3.0,];

# Ошибка: C-стиль объявления массива. Квадратные скобки должны быть частью типа.
# ПРАВИЛЬНО: int[] c
int c[] = [1, 2, 3];

# Ошибка: Отсутствие запятой между элементами массива.
int[] d = [1 2 3];


# --- Ошибки доступа к элементам и срезам ---

# Ошибка: Использование круглых скобок () для доступа к элементу.
# Для доступа к элементам должны использоваться квадратные скобки [].
int[] e = [10, 20];
int val = e(0);

# Ошибка: Использование точки с запятой ; вместо двоеточия : для создания среза.
int[] f = e[0; 1];


# --- Ошибки вызова методов ---

# Ошибка: Вызов .length как функции с (). .length является свойством.
# ПРАВИЛЬНО: g.length
int[] g = [1];
int len = g.length();

# Ошибка: Отсутствуют скобки при вызове метода .append().
g.append 100;
