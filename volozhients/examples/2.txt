node A;
node B;
node C;
node D;
node E;

arc <A,B>;
arc <A,C>;
arc <B,D>;
arc <C,E>;

graph G;
G = G * (A,B,C,D,E);
G = G + <A,B> + <A,C> + <B,D> + <C,E>;

print(BFS(G,A), "\n");
print(DFS(G,A), "\n");

void delNode(graph G, node B) {
    if (G.hasNode(B)) {
        G = G / B;
    }
}

delNode(G,B);

print(G, "%d");
