# ? - параметр результата, а паарметры результата раполагаются в конце

function factorial(integer ?n, integer result) {
    if (n <= 1) {
        result = 1;
    } else {
        result = n * factorial(n-1);
    }
}

integer input = in("Введите число для вычисления факториала: ");

integer result;

factorial(input, result);

out("Факториал числа " + input + " = " + result);
