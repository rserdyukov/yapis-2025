function scopeCheck() {
    integer x = 10;
    out("x до выполнения цикла в функици: " + x);

    for (integer i = 0; i <= 10; i++) {
        integer x = 5;
        out("x во время выполнения цикла в функции: " + x);
    }

    out("x после выполнения цикла в функции: " + x);
}

integer x = 20;
out("x до выполнения функции: " + x);
scopeCheck();
out("x после выполнения цикла: " + x);
