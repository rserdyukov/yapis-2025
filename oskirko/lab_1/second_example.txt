function fibonacci(integer n, integer ?result) {
    integer[n] numbers = [];
    numbers[0] = 0;
    numbers[1] = 1;
    
    for (integer i = 2; i < n; i++) {
        numbers[i] = numbers[i-1] + numbers[i-2];
    }

    result = numbers[n-1]; 
}

integer input = in("Введите число для вычисления определенного числа массива чисел Фибоначи: ");

integer result;

fibonacci(input, result);

out("Число Фибоначи на индексе " + input + " = " + result);
