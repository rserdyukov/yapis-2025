# Перегрузка функций запрещена

function factorial(integer n, integer ?result) {
    if (n <= 1) {
        result = 1;
    } else {
        integer nextFactorial;
        factorial(n-1, nextFactorial);
        result = n * nextFactorial;
    }
}
